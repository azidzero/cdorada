SELECT a.id_Ticketlinea AS id, a.Producto, a.Precio, 0, a.Cantidad, c.CantidadPagada,
(a.Cantidad * a.Precio) AS Total FROM ticketlineas a 
INNER JOIN tickets b ON (a.id_ticket = b.id_ticket)
INNER JOIN pedidoslineas c ON (b.idpedido = c.idpedido)
WHERE b.Id_Mesa = 120
AND b.impreso = 'SI'
AND c.idpedido = 3684;

SELECT a.idpedido AS ID, a.Articulos, b.Costo AS Precio, 
a.CostoPromo, (a.Cantidad - a.CantidadPagada) AS Cantidad, 
((a.Cantidad - a.CantidadPagada) * b.Costo) AS Total FROM pedidoslineas a
INNER JOIN articulos b ON (a.IdArticulo = b.Id_Articulos)
INNER JOIN pedidos c ON (a.idpedido = c.idpedidos)
WHERE c.id_mesa = 120
AND c.Pagado = 'NO';
